Carga el modelo más reciente y acepta desafíos. Por defecto se conecta al servidor local (localhost:8000), el mismo donde entrenas. No necesita login (el flag --no-security del servidor desactiva la autenticación).

Abre http://localhost:8000 en el navegador, elige el formato # correspondiente y desafía a "MiBot".

 55def _safe_mask(raw_mask, length: int, is_doubles: bool) -> np.ndarray:
 56    """Convierte la máscara de poke-env a bool y garantiza >=1 acción válida por hueco."""
 57    mask = np.asarray(raw_mask, dtype=bool)
 58    if mask.shape[0] != length:
 59        return np.ones(length, dtype=bool)
 60    if is_doubles:
 61        half = length // 2
 62        if not mask[:half].any():
 63            mask[0] = True
 64        if not mask[half:].any():
 65            mask[half] = True
 66    elif not mask.any():
 67        mask[0] = True
 68    return mask

 71class RLSinglesPlayer(Player):
 72    """
 73    Jugador singles que usa un modelo MaskablePPO cargado.
 74    Convierte el estado de batalla a observación, calcula la máscara de
 75    acciones legales y elige (determinísticamente) una acción legal.
 76    """
 77
 78    def __init__(self, model: MaskablePPO, *args, **kwargs):
 79        super().__init__(*args, **kwargs)
 80        self._model = model
 81
 82    def choose_move(self, battle):
 83        from poke_env.battle.battle import Battle
 84        from poke_env.environment.singles_env import SinglesEnv
 85        assert isinstance(battle, Battle)
 86        obs = embed_singles_battle(battle)
 87        try:
 88            mask = _safe_mask(
 89                SinglesEnv.get_action_mask(battle),
 90                SinglesEnv.get_action_space_size(battle.gen),
 91                is_doubles=False,
 92            )
 93            action, _ = self._model.predict(
 94                obs, deterministic=True, action_masks=mask
 95            )
 96            order = SinglesEnv.action_to_order(
 97                np.int64(action), battle, fake=False, strict=False
 98            )
 99        except Exception:
100            return self.choose_random_move(battle)
101        return order
102
103    def teampreview(self, battle):
104        return "/team 1234"

139def find_latest_model(model_dir: Path) -> Path | None:
140    """Busca el checkpoint más reciente en un directorio."""
141    checkpoints = sorted(
142        model_dir.glob("*.zip"),
143        key=lambda p: p.stat().st_mtime,
144    )
145    return checkpoints[-1] if checkpoints else None

148async def play_singles(
149    username: str,
150    password: str | None,
151    run_id: int,
152    n_games: int,
153    play_on_ladder: bool,
154    battle_format: str,
155    port: int,
156    official: bool,
157    policy: str = "mlp",
158    max_concurrent: int = 1,
159    tag: str = "",
160):
161    seed_dir = f"seed{run_id}" + (f"_{tag}" if tag else "")
162    model_dir = Path("models/singles") / battle_format / policy / seed_dir
163    model_path = find_latest_model(model_dir)
164    if model_path is None:
165        print(f"No se encontró ningún modelo en {model_dir}")
166        print(f"Entrena primero con: python train_singles.py --format {battle_format} --policy {policy}")
167        return
168
169    print(f"Cargando modelo desde {model_path}")
170    model = MaskablePPO.load(str(model_path))
171
172    server = "OFICIAL" if official else f"LOCAL (localhost:{port})"
173    print(f"Conectando al servidor {server}")
174    player = RLSinglesPlayer(
175        model=model,
176        account_configuration=AccountConfiguration(username, password),
177        battle_format=battle_format,
178        server_configuration=make_server_config(port, official),
179        log_level=20,
180        max_concurrent_battles=max_concurrent,
181    )
182
183    if play_on_ladder:
184        print(f"Entrando en la ladder ({n_games} partidas)...")
185        await player.ladder(n_games=n_games)
186        print(
187            f"Resultado: {player.n_won_battles}W - "
188            f"{player.n_lost_battles}L - {player.n_tied_battles}T"
189        )
190    else:
191        print(f"Esperando {n_games} desafíos en {battle_format}...")
192        await player.accept_challenges(opponent=None, n_challenges=n_games)
193        print(
194            f"Resultado: {player.n_won_battles}W - "
195            f"{player.n_lost_battles}L - {player.n_tied_battles}T"
196        )

Returns a teampreview order for the given battle.

This order must be of the form /team TEAM, where TEAM is a string defining the team chosen by the player. Multiple formats are supported, among which '3461' and '3, 4, 6, 1' are correct and indicate leading with pokemon 3, with pokemons 4, 6 and 1 in the back in single battles or leading with pokemons 3 and 4 with pokemons 6 and 1 in the back in double battles.

Please refer to Pokemon Showdown's protocol documentation for more information.

IMPORTANT: If overwriting this method, make sure to mark Pokemon selected in teampreview as such by setting their _selected_in_teampreview attribute to True. See random_teampreview implementation for an example.

Parameters
  • battle: The battle.
Returns

The teampreview order.
