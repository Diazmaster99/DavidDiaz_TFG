26class RandomTeamBuilder(Teambuilder):
27 """
28 Devuelve un equipo aleatorio en cada yield_team().
29
30 :param teams_dir: carpeta con ficheros .txt, cada uno un equipo en
31 formato Showdown (el que exporta el Teambuilder de la web).
32 """
33
34 def __init__(self, teams_dir: str | Path):
35 teams_dir = Path(teams_dir)
36 files = sorted(teams_dir.glob("*.txt"))
37 if not files:
38 raise FileNotFoundError(
39 f"No se encontraron ficheros .txt en {teams_dir}\n"
40 "Exporta equipos desde https://play.pokemonshowdown.com/teambuilder\n"
41 "y guarda cada uno como un .txt en esa carpeta."
42 )
43 self.team_files = files
44 self.packed_teams: list[str] = []
45 for f in files:
46 team_str = f.read_text(encoding="utf-8").strip()
47 if not team_str:
48 continue
49 mons = self.parse_showdown_team(team_str)
50 self.packed_teams.append(self.join_team(mons))
51 if not self.packed_teams:
52 raise ValueError(f"Todos los ficheros de {teams_dir} están vacios.")
53
54 def yield_team(self) -> str:
55 return random.choice(self.packed_teams)
56
57 def __len__(self) -> int:
58 return len(self.packed_teams)

61def build_teambuilder(path: str | Path | None):
62 """
63 Construye el teambuilder adecuado según la ruta:

65 - fichero .txt -> ConstantTeambuilder (siempre el mismo equipo)
66 - carpeta -> RandomTeamBuilder (equipo aleatorio por combate)
