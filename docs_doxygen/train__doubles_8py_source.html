50def checkpoint_steps(path) -> int | None:
51 """
52 Extrae el número de pasos del nombre de un checkpoint.
53 Devuelve None para ficheros sin número (por ejemplo 'final.zip').
54 Ejemplos: 'model_20000_steps' -> 20000 ; 'final' -> None
55 """
56 stem = path.stem
57 cleaned = stem.replace("model_", "").replace("_steps", "")
58 try:
59 return int(cleaned)
60 except ValueError:
61 return None

64def load_team(team_path: str | None) -> str | None:
65 """Carga un equipo desde fichero de texto en formato Showdown."""

95def train(
96 run_id: int,
97 port: int,
98 device: str,
99 battle_format: str,
100 team_path: str | None,
101 total_steps: int,
102 save_interval: int,
103 opponent_type: str = "random",
104 policy: str = "mlp",
105 tag: str = "",
106):
107 set_seed(run_id)
108 check_server(port)
109
110 # Validar compatibilidad formato <-> equipo
111 needs_team = "random" not in battle_format
112 if needs_team and team_path is None:
113 raise ValueError(
114 f"El formato '{battle_format}' requiere un equipo.\n"
115 f" Un equipo fijo: --team teams/vgc/equipo_ejemplo.txt\n"
116 f" Equipos variados: --team teams/vgc (carpeta con varios .txt)\n"
117 f" O usa formato aleatorio: --format gen9randomdoublesbattle"
118 )
119 if not needs_team and team_path is not None:
120 print(f"Nota: --team ignorado para formato de batalla aleatorio '{battle_format}'")
121 team_path = None
122
123 if team_path is not None and Path(team_path).is_dir():
124 n_teams = len(list(Path(team_path).glob("*.txt")))
125 print(f"Usando {n_teams} equipos de '{team_path}' (aleatorio por combate)")
126
127 # Separar modelos por formato Y por política (para comparar mlp vs transformer)
128 # 'tag' permite diferenciar modelos/runs (p.ej. tag="selfplay") sin mezclarlos
129 # con el mismo seed entrenado contra otro oponente.
130 seed_dir = f"seed{run_id}" + (f"_{tag}" if tag else "")
131 save_dir = Path("models/doubles") / battle_format / policy / seed_dir
132 log_dir = Path("logs/doubles") / battle_format / policy
133 save_dir.mkdir(parents=True, exist_ok=True)
134 log_dir.mkdir(parents=True, exist_ok=True)
135
136 print(f"Creando entorno doubles {battle_format} contra {opponent_type} (policy={policy})...")
137 # DoublesGymEnv crea su propio oponente (sin conexión al servidor)
138 env = DoublesGymEnv(
139 port=port,
140 battle_format=battle_format,
141 log_level=40,
142 team_path=team_path,
143 opponent_type=opponent_type,
144 )
145 print(f"Espacio de acción: {env.action_space}")
146 print(f"Espacio de observación: {env.observation_space}")
147
148 # Reanudar desde checkpoint si existe. Probamos de MAYOR a menor número de
149 # pasos y cargamos el primero compatible. Así se ignoran automáticamente los
150 # checkpoints antiguos de PPO sin action masking (no son cargables por
151 # MaskablePPO -> "Policy must subclass MaskableActorCriticPolicy").
152 num_saved = 0
153 model = None
154 numbered = sorted(
155 [
156 (p, checkpoint_steps(p))
157 for p in save_dir.glob("*.zip")
158 if checkpoint_steps(p) is not None
159 ],
160 key=lambda t: t[1],
161 reverse=True,
162 )
163 for cand, steps in numbered:
164 try:
165 model = MaskablePPO.load(str(cand), env=env, device=device)
166 model.num_timesteps = steps
167 num_saved = steps
168 print(f"Reanudando desde {cand} ({steps} pasos)")
169 break
170 except Exception as e:
171 print(f"AVISO: {cand.name} no es compatible ({type(e).__name__}); "
172 "probando un checkpoint anterior...")
173 model = None
174
175 if model is None:
176 from src.policy import make_policy_kwargs
177 print(f"Creando nuevo modelo MaskablePPO (policy={policy}, action masking)...")
178 model = MaskablePPO(
179 "MlpPolicy",
180 env,
181 learning_rate=lambda p: 3e-4 * p,
182 n_steps=2048,
183 batch_size=256,
184 n_epochs=10,
185 gamma=0.99,
186 gae_lambda=0.95,
187 clip_range=0.2,
188 ent_coef=0.01,
189 policy_kwargs=make_policy_kwargs(policy),
190 verbose=1,
191 tensorboard_log=str(log_dir),
192 device=device,
193 )
194
195 # SELF-PLAY: el rival juega con la propia política que se está entrenando.
196 if opponent_type == "self":
197 env.set_opponent_model(model)
198 print("SELF-PLAY activado: el rival usa la política actual del agente.")
199
200 from src.callbacks import WinRateCallback
201 checkpoint_cb = CheckpointCallback(
202 save_freq=save_interval,
203 save_path=str(save_dir),
204 name_prefix="model",
205 verbose=1,
206 )
207 # Registra eval/win_rate en TensorBoard (winrate móvil de las últimas batallas)
208 winrate_cb = WinRateCallback(log_freq=5000, window=200)
209
210 remaining = total_steps - num_saved
211 if remaining <= 0:
212 print(f"Ya se han completado {num_saved} >= {total_steps} pasos.")
213 else:
214 print(f"Entrenando {remaining} pasos...")
215 model.learn(
216 total_timesteps=remaining,
217 callback=[checkpoint_cb, winrate_cb],
218 reset_num_timesteps=False,
219 tb_log_name=f"doubles_seed{run_id}" + (f"_{tag}" if tag else ""),
220 )
221 final_path = save_dir / "final"
222 model.save(str(final_path))
223 print(f"Modelo final guardado en {final_path}.zip")
224
225 env.close()
226 print("Entrenamiento completado.")
