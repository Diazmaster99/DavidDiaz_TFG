55 def calc_reward(self, battle: AbstractBattle) -> float:
56 """
57 Recompensa:
58 +1.0 al ganar, -1.0 al perder.
59 Pequenio bonus intermedio basado en HP relativo.
60 """
61 assert isinstance(battle, Battle)
62 if battle.finished:
63 return 1.0 if battle.won else -1.0
64
65 # Recompensa intermedia: diferencia de HP para dar gradiente
66 own_hp = sum(
67 m.current_hp_fraction for m in battle.team.values()
68 ) / max(len(battle.team), 1)
69 opp_hp = sum(
70 m.current_hp_fraction for m in battle.opponent_team.values()
71 ) / max(len(battle.opponent_team), 1)
72 return (own_hp - opp_hp) * 0.01
