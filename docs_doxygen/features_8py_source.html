5"""
6Construcción de observaciones detalladas para los entornos (estilo vgc-bench).
7
8La idea es dar al agente RL toda la información que tendría un jugador humano
9(efectividades, clima, campo, prioridades, estados, boosts, stats, side
10conditions, etc.) y dejar que la red aprenda sola qué hacer.
11
12Se exponen dos funciones:
13 build_singles_obs(battle) -> np.ndarray (SINGLES_OBS_SIZE,)
14 build_doubles_obs(battle) -> np.ndarray (DOUBLES_OBS_SIZE,)
15
16Diseño por bloques de tamaño FIJO (rellenando con ceros lo que no exista),
17de modo que la observación siempre tiene la misma longitud.
18"""

54N_TYPES = len(TYPE_NAMES) # 18

57ACTIVE_FEATS = 3 + len(STATUS_NAMES) + N_TYPES + len(BOOST_KEYS) + len(STAT_KEYS) # 40
58RESERVE_FEATS = 2 + len(STATUS_NAMES) + N_TYPES # 26

76def singles_token_segments() -> list[tuple[str, int, int]]:
77 """
78 Layout del vector plano de singles como secuencia de TOKENS (kind, inicio, len).
79 DEBE coincidir con el orden en que build_singles_obs concatena los bloques.
80 Lo usa el extractor transformer para trocear la observación en entidades.
81 """
82 segs: list[tuple[str, int, int]] = []
83 off = 0
84 g = GLOBAL_FEATS + 2 * SIDE_FEATS # global + 2 bandos = 1 token
85 segs.append(("global", off, g)); off += g
86 for _ in range(2): # activo propio, activo rival
87 segs.append(("active", off, ACTIVE_FEATS)); off += ACTIVE_FEATS
88 for _ in range(2 * SINGLES_BENCH): # banco propio (5) + rival (5)
89 segs.append(("reserve", off, RESERVE_FEATS)); off += RESERVE_FEATS
90 for _ in range(4): # 4 movimientos del activo propio
91 segs.append(("move", off, MOVE_FEATS_SINGLES)); off += MOVE_FEATS_SINGLES
92 assert off == SINGLES_OBS_SIZE, f"{off} != {SINGLES_OBS_SIZE}"
93 return segs

96def doubles_token_segments() -> list[tuple[str, int, int]]:
97 """Layout de tokens de dobles. Coincide con build_doubles_obs."""
98 segs: list[tuple[str, int, int]] = []
99 off = 0
100 g = GLOBAL_FEATS + 2 * SIDE_FEATS
101 segs.append(("global", off, g)); off += g
102 for _ in range(4): # 2 activos propios + 2 rivales
103 segs.append(("active", off, ACTIVE_FEATS)); off += ACTIVE_FEATS
104 for _ in range(2 * DOUBLES_BENCH): # banco propio (4) + rival (4)
105 segs.append(("reserve", off, RESERVE_FEATS)); off += RESERVE_FEATS
106 for _ in range(8): # 4 movimientos x 2 activos propios
107 segs.append(("move", off, MOVE_FEATS_DOUBLES)); off += MOVE_FEATS_DOUBLES
108 assert off == DOUBLES_OBS_SIZE, f"{off} != {DOUBLES_OBS_SIZE}"
109 return segs

116def _active_names(container) -> set[str]:
117 """Nombres de los enums activos en un dict/set/enum/None (clima, campo...)."""
118 if not container:
119 return set()
120 if isinstance(container, dict):
121 return {k.name for k in container}
122 if isinstance(container, (set, list, tuple)):
123 return {k.name for k in container}
124 return {container.name}

127def _coerce_float(value, default: float = 0.0) -> float:
128 if value is True:
129 return 1.0
130 if value is False or value is None:
131 return default
132 try:
133 return float(value)
134 except (TypeError, ValueError):
135 return 1.0 # p.ej. self_switch = "copyvolatile"

138def _type_multihot(mon) -> list[float]:
139 vec = [0.0] * N_TYPES
140 if mon is not None:
141 for t in mon.types:
142 if t is not None and t.name in TYPE_IDX:
143 vec[TYPE_IDX[t.name]] = 1.0
144 return vec

147def _status_onehot(mon) -> list[float]:
148 vec = [0.0] * len(STATUS_NAMES)
149 if mon is not None and mon.status is not None and mon.status.name in STATUS_NAMES:
150 vec[STATUS_NAMES.index(mon.status.name)] = 1.0

166def _active_features(mon) -> list[float]:
167 """Bloque de un Pokémon ACTIVO (propio o rival). Longitud = ACTIVE_FEATS."""
168 if mon is None:
169 return [0.0] * ACTIVE_FEATS
170 feats = [
171 mon.current_hp_fraction or 0.0,
172 1.0 if mon.fainted else 0.0,
173 1.0 if mon.is_terastallized else 0.0,
174 ]
175 feats += _status_onehot(mon)
176 feats += _type_multihot(mon)
177 feats += _boosts(mon)
178 feats += _base_stats(mon)
179 return feats

183 """Bloque de un Pokémon en el BANCO. Longitud = RESERVE_FEATS."""
184 if mon is None:

195def _move_features(move, opponents) -> list[float]:
196 """
197 Bloque de un movimiento + su efectividad contra cada rival de 'opponents'.
198 Longitud = _MOVE_SCALARS + 3 (categoría) + N_TYPES + len(opponents).
199 """
200 n_opp = len(opponents)
201 if move is None:
202 return [0.0] * (_MOVE_SCALARS + 3 + N_TYPES + n_opp)
203
204 acc = _coerce_float(move.accuracy, 1.0)
205 max_pp = move.max_pp or 1
206 pp_frac = (move.current_pp or 0) / max_pp
207 scalars = [
208 (move.base_power or 0) / 100.0,
209 acc,
210 (move.priority or 0) / 5.0,
211 (move.crit_ratio or 0) / 6.0,
212 _coerce_float(move.drain),
213 _coerce_float(move.recoil),
214 1.0 if move.force_switch else 0.0,
215 _coerce_float(move.self_switch),
216 1.0 if move.self_destruct is not None else 0.0,
217 pp_frac,
218 (move.expected_hits or 1) / 5.0,
219 ]
220 # Categoría one-hot: PHYSICAL, SPECIAL, STATUS
221 category = [0.0, 0.0, 0.0]
222 cat_idx = {"PHYSICAL": 0, "SPECIAL": 1, "STATUS": 2}.get(move.category.name)
223 if cat_idx is not None:
224 category[cat_idx] = 1.0
225 # Tipo del movimiento one-hot
226 move_type = [0.0] * N_TYPES
227 if move.type is not None and move.type.name in TYPE_IDX:
228 move_type[TYPE_IDX[move.type.name]] = 1.0
229 # Efectividad de tipo contra cada rival
230 eff = []
231 for opp in opponents:
232 if opp is None or move.type is None:
233 eff.append(0.0)
234 else:
235 eff.append(
236 move.type.damage_multiplier(
237 opp.type_1, opp.type_2, type_chart=_CHART
238 )
239 )
240 return scalars + category + move_type + eff

243def _moves_block(active_mon, opponents, move_feat_len: int) -> list[float]:
244 """4 movimientos del Pokémon activo (rellenando con ceros si faltan)."""
245 block: list[float] = []
246 moves = []
247 if active_mon is not None:
248 moves = list(active_mon.moves.values())[:4]
249 for i in range(4):
250 mv = moves[i] if i < len(moves) else None
251 block += _move_features(mv, opponents)
252 assert len(block) == 4 * move_feat_len
253 return block

256def _global_features(battle) -> list[float]:
257 feats: list[float] = []
258 w = _active_names(battle.weather)
259 feats += [1.0 if name in w else 0.0 for name in WEATHER_NAMES]
260 f = _active_names(battle.fields)
261 feats += [1.0 if name in f else 0.0 for name in FIELD_NAMES]
262 feats.append(min(battle.turn / 20.0, 2.0))
263 return feats

266def _side_features(conditions) -> list[float]:
267 active = _active_names(conditions)
268 return [1.0 if name in active else 0.0 for name in SIDE_NAMES]

275def build_singles_obs(battle: Battle) -> npt.NDArray[np.float32]:
276 own = battle.active_pokemon
277 opp = battle.opponent_active_pokemon
278
279 feats: list[float] = []
280 feats += _global_features(battle) # 15
281 feats += _side_features(battle.side_conditions) # 10
282 feats += _side_features(battle.opponent_side_conditions) # 10
283 feats += _active_features(own) # 40
284 feats += _active_features(opp) # 40
285
286 own_bench = [m for m in battle.team.values() if m is not own][:SINGLES_BENCH]
287 for i in range(SINGLES_BENCH):
288 feats += _reserve_features(own_bench[i] if i < len(own_bench) else None)
289 opp_bench = [
290 m for m in battle.opponent_team.values() if m is not opp
291 ][:SINGLES_BENCH]
292 for i in range(SINGLES_BENCH):
293 feats += _reserve_features(opp_bench[i] if i < len(opp_bench) else None)
294
295 feats += _moves_block(own, [opp], MOVE_FEATS_SINGLES) # 4*33
296
297 assert len(feats) == SINGLES_OBS_SIZE, f"{len(feats)} != {SINGLES_OBS_SIZE}"
298 return np.array(feats, dtype=np.float32)

301def build_doubles_obs(battle: DoubleBattle) -> npt.NDArray[np.float32]:
302 own_active = list(battle.active_pokemon) + [None, None]
303 opp_active = list(battle.opponent_active_pokemon) + [None, None]
304 own0, own1 = own_active[0], own_active[1]
305 opps = [opp_active[0], opp_active[1]]
306
307 feats: list[float] = []
308 feats += _global_features(battle) # 15
309 feats += _side_features(battle.side_conditions) # 10
310 feats += _side_features(battle.opponent_side_conditions) # 10
311 feats += _active_features(own0) # 40
312 feats += _active_features(own1) # 40
313 feats += _active_features(opps[0]) # 40
314 feats += _active_features(opps[1]) # 40
315
316 active_set = {id(m) for m in (own0, own1) if m is not None}
317 own_bench = [
318 m for m in battle.team.values() if id(m) not in active_set
319 ][:DOUBLES_BENCH]
320 for i in range(DOUBLES_BENCH):
321 feats += _reserve_features(own_bench[i] if i < len(own_bench) else None)
322 opp_active_set = {id(m) for m in opps if m is not None}
323 opp_bench = [
324 m for m in battle.opponent_team.values() if id(m) not in opp_active_set
325 ][:DOUBLES_BENCH]
326 for i in range(DOUBLES_BENCH):
327 feats += _reserve_features(opp_bench[i] if i < len(opp_bench) else None)
328
329 # Movimientos de cada activo propio, con efectividad vs los 2 rivales
330 feats += _moves_block(own0, opps, MOVE_FEATS_DOUBLES) # 4*34
331 feats += _moves_block(own1, opps, MOVE_FEATS_DOUBLES) # 4*34
332
333 assert len(feats) == DOUBLES_OBS_SIZE, f"{len(feats)} != {DOUBLES_OBS_SIZE}"
334 return np.array(feats, dtype=np.float32)
