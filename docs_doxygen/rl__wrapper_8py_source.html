5"""
6Wrapper Gymnasium que controla a un agente con RL y al oponente, con
7RECUPERACIÓN ROBUSTA ante los cuelgues conocidos de poke-env + servidor local.
8
9Doble función:
10 1) Convierte el PokeEnv (PettingZoo, 2 agentes) en un entorno Gymnasium de un
11 solo agente. agent1 = política RL; agent2 = oponente (aleatorio válido,
12 heurístico, o self-play con una política).
13 2) Hace el entrenamiento RESILIENTE:
14 - watchdog en step(): si un turno se queda colgado (desincronización
15 poke-env/servidor en dobles), lo abandona en 'step_timeout' segundos
16 en vez de esperar minutos al temporizador del servidor.
17 - reconstrucción del entorno: si reset() o un turno fallan, se cierra la
18 conexión rota y se crea un PokeEnv NUEVO (agentes/websocket limpios),
19 que es lo único que recupera de un estado de conexión corrupto.
20
21NOTA: el oponente se muestrea de la máscara de acciones (no se usa el
22SingleAgentWrapper de poke-env, que tiene un bug en order_to_action de dobles).
23"""

54 def __init__(self, env_factory, opponent=None, step_timeout: float = 30.0):
55 """
56 :param env_factory: callable SIN argumentos que crea un PokeEnv listo
57 (ya configurado, con sus filtros de log). Se usa para crear el
58 entorno y para RECONSTRUIRLO si la conexión se rompe.
59 :param opponent: Player opcional (p.ej. SimpleHeuristicsPlayer) para
60 decidir las jugadas del rival. None -> aleatorio válido (o self-play
61 si se asigna un modelo con set_opponent_model()).
62 :param step_timeout: segundos máximos para un turno antes de considerar
63 el combate atascado y reconstruir el entorno.
64 """

66 self._env = env_factory()
67 self._opponent = opponent
68 self._opponent_model = None # SELF-PLAY si se asigna
69 self._step_timeout = step_timeout
70 self._executor = ThreadPoolExecutor(max_workers=1)

74 self.action_space = self._env.action_spaces[a1]

77 self._mask_len = int(self.action_space.nvec.sum()) # 214
78 else:

95 def _force_finish_battles(self):
96 """Marca como terminadas las batallas colgadas en los 2 agentes."""
97 pe = self._env
98 for agent in (getattr(pe, "agent1", None), getattr(pe, "agent2", None)):
99 if agent is None:
100 continue
101 try:
102 for b in list(agent._battles.values()):
103 if not b.finished:
104 b._finished = True
105 except Exception:
106 pass
107 pe.battle1 = None
108 pe.battle2 = None

110 def _finish_lingering_battles(self):
111 """Marca como terminadas las batallas locales que hayan quedado sin
112 cerrar (p.ej. tras un turno abortado, o un final no detectado en uno de
113 los dos bandos en dobles). Así poke-env puede reiniciar sin lanzar
114 'Can not reset player's battles while they are still running', evitando
115 la costosa reconstrucción del entorno. No anula battle1/battle2: poke-env
116 los reasigna en su propio reset()."""
117 pe = self._env
118 for agent in (getattr(pe, "agent1", None), getattr(pe, "agent2", None)):
119 if agent is None:
120 continue
121 try:
122 for b in list(agent._battles.values()):
123 if not b.finished:
124 b._finished = True
125 except Exception:
126 pass

128 def _rebuild(self):
129 """Cierra el entorno roto y crea uno NUEVO (conexión/agentes limpios)."""
130 self._force_finish_battles()
131 try:
132 # IMPORTANTE: wait=False. El close() de poke-env, con su valor por
133 # defecto wait=True, hace self._challenge_task.result() SIN timeout;
134 # si el desafío está colgado (justo lo que provoca "Agent is not
135 # challenging"), ese .result() bloquea PARA SIEMPRE y congela todo el
136 # entrenamiento. Con wait=False se descarta la tarea colgada y se crea
137 # un entorno nuevo y limpio.
138 self._env.close(wait=False)
139 except Exception:
140 pass
141 # El worker del executor puede estar bloqueado en un step colgado:
142 # lo abandonamos y creamos uno nuevo.
143 try:
144 self._executor.shutdown(wait=False, cancel_futures=True)
145 except Exception:
146 pass
147 self._executor = ThreadPoolExecutor(max_workers=1)
148 time.sleep(2)
149 self._env = self._factory()

153 def reset(self, *, seed=None, options=None):
154 attempts = 4
155 for i in range(attempts):
156 try:
157 # Limpieza preventiva: cierra batallas locales sin terminar para
158 # evitar el OSError "battles still running" y la reconstrucción
159 # del entorno (que es lenta). Es inocua en el reset normal.
160 self._finish_lingering_battles()
161 obs, infos = self._env.reset(seed, options)
162 if self._opponent is not None and self._env.battle2 is not None:
163 self._opponent.reset_battles()
164 self._opponent._battles[
165 self._env.battle2.battle_tag
166 ] = self._env.battle2
167 a1 = self._env.agents[0]
168 self._last_mask = self._extract_mask(obs[a1])
169 return obs[a1]["observation"], infos[a1]
170 except (TimeoutError, OSError) as e:
171 # TimeoutError ("Agent is not challenging") y OSError ("battles
172 # still running") indican conexión/estado roto: reconstruimos.
173 if DEBUG:
174 # Modo debug: sin reintentos; muestra el traceback real.
175 print("[env.reset] MODO DEBUG: traceback real ->", flush=True)
176 traceback.print_exc()
177 raise
178 if i == attempts - 1:
179 raise
180 print(
181 f"[env.reset] fallo ({type(e).__name__}: {e}); reconstruyendo "
182 f"entorno y reintentando ({i + 1}/{attempts - 1})...",
183 flush=True,
184 )
185 self._rebuild()

187 def step(self, action):
188 pe = self._env
189 b1, b2 = pe.battle1, pe.battle2
190
191 # Guardia: si la batalla ya terminó, estado terminal limpio.
192 if (b1 is not None and b1.finished) or (b2 is not None and b2.finished):
193 reward = 0.0
194 if b1 is not None and b1.finished:
195 reward = 1.0 if b1.won else -1.0
196 pe.battle1 = None
197 pe.battle2 = None
198 self._last_mask = np.ones(self._mask_len, dtype=bool)
199 return self._zeros(), reward, True, False, {}
200
201 a1, a2 = pe.agents[0], pe.agents[1]
202 opp_action = self._opponent_action(b2)
203 actions = {a1: action, a2: opp_action}
204
205 # Ejecutar el turno con WATCHDOG: si tarda más de step_timeout, el
206 # combate está atascado -> abandonar y reconstruir (rápido, sin esperar
207 # al temporizador del servidor, que tarda minutos).
208 if DEBUG:
209 # Modo debug: sin watchdog ni captura -> traceback real del turno.
210 obs, rewards, terms, truncs, infos = pe.step(actions)
211 else:
212 try:
213 fut = self._executor.submit(pe.step, actions)
214 obs, rewards, terms, truncs, infos = fut.result(timeout=self._step_timeout)
215 except FuturesTimeoutError:
216 print(
217 f"[env.step] turno atascado >{self._step_timeout:.0f}s; "
218 f"abandonando combate y reconstruyendo entorno...",
219 flush=True,
220 )
221 self._rebuild()
222 self._last_mask = np.ones(self._mask_len, dtype=bool)
223 return self._zeros(), 0.0, True, False, {}
224 except (AssertionError, ValueError, IndexError):
225 pe.battle1 = None
226 pe.battle2 = None
227 self._last_mask = np.ones(self._mask_len, dtype=bool)
228 return self._zeros(), 0.0, True, False, {}
229
230 a1_obs = obs[a1]
231 self._last_mask = self._extract_mask(a1_obs)
232 return (
233 a1_obs["observation"],
234 rewards[a1],
235 terms[a1],
236 truncs[a1],
237 infos[a1],
238 )

246 def _extract_mask(self, agent_obs) -> np.ndarray:
247 return self._sanitize_mask(agent_obs["action_mask"])

249 def _sanitize_mask(self, mask) -> np.ndarray:
250 mask = np.asarray(mask, dtype=bool)
251 if mask.shape[0] != self._mask_len:
252 return np.ones(self._mask_len, dtype=bool)
253 if self._is_doubles:
254 half = self._mask_len // 2
255 if not mask[:half].any():
256 mask[0] = True
257 if not mask[half:].any():
258 mask[half] = True
259 elif not mask.any():
260 mask[0] = True
261 return mask

266 def _opponent_action(self, battle):
267 """
268 Acción del oponente, por prioridad:
269 1. SELF-PLAY (opponent_model) -> política RL.
270 2. HEURÍSTICO (Player) -> choose_move.
271 3. ALEATORIO -> muestreo de la máscara.
272 """
273 if self._opponent_model is not None:
274 try:
275 obs = self._env.embed_battle(battle)
276 mask = self._sanitize_mask(self._env.get_action_mask(battle))
277 action, _ = self._opponent_model.predict(
278 obs, action_masks=mask, deterministic=False
279 )
280 action = np.asarray(action)
281 if self._is_doubles:
282 return action.reshape(-1)[:2].astype(np.int64)
283 return np.int64(action.reshape(-1)[0])
284 except Exception:
285 if DEBUG:
286 print("[opponent_model] error (se usaría aleatorio) ->", flush=True)
287 traceback.print_exc()
288
289 if self._opponent is not None:
290 try:
291 from collections.abc import Awaitable
292 order = self._opponent.choose_move(battle)
293 if not isinstance(order, Awaitable):
294 return self._env.order_to_action(
295 order, battle, fake=False, strict=False
296 )
297 except Exception:
298 if DEBUG:
299 print("[opponent] error (se usaría aleatorio) ->", flush=True)
300 traceback.print_exc()
301
302 return self._sample_from_mask(battle)
303
304 def _sample_from_mask(self, battle):
305 mask = self._env.get_action_mask(battle)
306 if self._is_doubles:
307 half = len(mask) // 2
308 valid0 = [i for i, m in enumerate(mask[:half]) if m] or [0]
309 valid1 = [i for i, m in enumerate(mask[half:]) if m] or [0]
310 return np.array(
311 [random.choice(valid0), random.choice(valid1)], dtype=np.int64
312 )
313 valid = [i for i, m in enumerate(mask) if m] or [0]
314 return np.int64(random.choice(valid))
