5"""
6Arranca un entrenamiento de SELF-PLAY a partir de un modelo YA ENTRENADO.
7
8Copia el último checkpoint de un modelo de origen a un seed nuevo marcado con un
9sufijo (por defecto 'selfplay'), de modo que:
10 - el modelo de self-play no sobrescribe al de origen (vs random/heurístico),
11 - la carpeta y el run de TensorBoard quedan claramente diferenciados
12 (p.ej. models/doubles/<formato>/mlp/seed1_selfplay/ y el run
13 'doubles_seed1_selfplay'),
14 - el entrenamiento se reanuda desde ese checkpoint y continúa con --opponent self.
15
16Uso típico (dobles, MLP, partiendo del modelo de seed1, formato VGC):
17 python selfplay_from.py --mode doubles --format gen9vgc2026regi --policy mlp \\
18 --src_seed 1 --team teams/vgc/reg_i --total_steps 1500000
19
20Solo copiar y mostrar el comando (sin lanzar):
21 python selfplay_from.py --mode doubles --src_seed 1 --no-launch
22
23Requisitos para lanzar: servidor Showdown corriendo (lo comprueba train_*.py).
24"""

34def latest_numbered_checkpoint(folder: Path):
35 """Devuelve (ruta, pasos) del checkpoint numerado más avanzado, o (None, -1)."""
36 best, best_steps = None, -1
37 for p in folder.glob("model_*_steps.zip"):
38 m = re.search(r"model_(\d+)_steps", p.stem)
39 if m and int(m.group(1)) > best_steps:
40 best, best_steps = p, int(m.group(1))
41 return best, best_steps
