5"""
6Extractor de características basado en TRANSFORMER (atención)
7
8En vez de tratar la observación como un vector plano (lo que hace el MLP),
9la trocea en TOKENS (una "ficha" por entidad: global, cada Pokémon, cada
10movimiento), proyecta cada ficha a una dimensión común y aplica atención para
11que las fichas se "miren" entre sí. Un token CLS resume toda la batalla.
12
13Ventajas frente al MLP plano:
14 - Un MISMO codificador por tipo de entidad (peso compartido) -> generaliza
15 mejor (lo aprendido de un Pokémon sirve para todos).
16 - La atención captura relaciones (matchups, sinergias) de forma natural.
17
18Se activa con --policy transformer en los scripts de entrenamiento. El MLP
19sigue siendo el valor por defecto (para comparar ambos en la memoria).
20"""
21

37class BattleTransformerExtractor(BaseFeaturesExtractor):
38 """
39 Convierte la observación plana en una secuencia de tokens y aplica un
40 TransformerEncoder. Devuelve el embedding del token CLS como vector de
41 características (sobre el que SB3 monta las cabezas de política/valor).
42
43 :param embed_dim: dimensión de cada token (d_model del transformer).
44 :param nhead: número de cabezas de atención.
45 :param num_layers: número de capas del encoder.
46 :param dim_ff: dimensión de la capa feed-forward interna.
47 """
48
49 def __init__(
50 self,
51 observation_space: spaces.Box,
52 embed_dim: int = 64,
53 nhead: int = 4,
54 num_layers: int = 2,
55 dim_ff: int = 128,
56 ):
57 super().__init__(observation_space, features_dim=embed_dim)
58 n = int(observation_space.shape[0])
59 if n == SINGLES_OBS_SIZE:
60 self._segments = singles_token_segments()
61 elif n == DOUBLES_OBS_SIZE:
62 self._segments = doubles_token_segments()
63 else:
64 raise ValueError(
65 f"Observación de dim {n} no soportada por el transformer "
66 f"(esperado {SINGLES_OBS_SIZE} o {DOUBLES_OBS_SIZE})."
67 )
68
69 # Una proyección lineal por TIPO de entidad (peso compartido entre
70 # todas las fichas del mismo tipo: todos los Pokémon usan el mismo
71 # encoder, todos los movimientos el mismo, etc.).
72 lengths: dict[str, int] = {}
73 for kind, _, length in self._segments:
74 lengths[kind] = length
75 self.proj = nn.ModuleDict(
76 {kind: nn.Linear(length, embed_dim) for kind, length in lengths.items()}
77 )
78
79 # Token CLS aprendible que resumirá toda la batalla.
80 self.cls = nn.Parameter(torch.zeros(1, 1, embed_dim))
81 nn.init.normal_(self.cls, std=0.02)
82
83 encoder_layer = nn.TransformerEncoderLayer(
84 d_model=embed_dim,
85 nhead=nhead,
86 dim_feedforward=dim_ff,
87 dropout=0.0,
88 batch_first=True,
89 activation="gelu",
90 )
91 self.encoder = nn.TransformerEncoder(encoder_layer, num_layers=num_layers)
92
93 def forward(self, observations: torch.Tensor) -> torch.Tensor:
94 batch = observations.shape[0]
95 tokens = []
96 for kind, start, length in self._segments:
97 segment = observations[:, start : start + length] # (B, length)
98 tokens.append(self.proj[kind](segment)) # (B, embed_dim)
99 x = torch.stack(tokens, dim=1) # (B, n_tokens, D)
100 cls = self.cls.expand(batch, -1, -1) # (B, 1, D)
101 x = torch.cat([cls, x], dim=1) # (B, 1+n_tokens, D)
102 x = self.encoder(x) # (B, 1+n_tokens, D)
103 return x[:, 0] # embedding del CLS
