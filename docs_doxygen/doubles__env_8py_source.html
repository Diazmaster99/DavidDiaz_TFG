37class _SuppressOpenTeamSheetFilter(logging.Filter):
38 """
39 Descarta el mensaje CRITICAL inofensivo de poke-env:
40 "This format does not allow requesting open team sheets."
41
42 poke-env envia automáticamente /rejectopenteamsheets en cualquier formato
43 VGC no-Bo3, pero algunos formatos (p.ej. gen9vgc2026regi no-Bo3) no permiten
44 esa peticion y el servidor la rechaza. La batalla continua sin problemas;
45 solo silenciamos el ruido en el log.
46 """
47
48 def filter(self, record: logging.LogRecord) -> bool:
49 return "open team sheets" not in record.getMessage().lower()

75 def calc_reward(self, battle: AbstractBattle) -> float:
76 """
77 Recompensa:
78 +1.0 al ganar, -1.0 al perder.
79 Pequenio bonus de HP intermedio.
80 """
81 assert isinstance(battle, DoubleBattle)
82 if battle.finished:
83 return 1.0 if battle.won else -1.0
84
85 # Bonus intermedio basado en ventaja de HP
86 own_hp = sum(
87 m.current_hp_fraction for m in battle.team.values()
88 ) / max(len(battle.team), 1)
89 opp_hp = sum(
90 m.current_hp_fraction for m in battle.opponent_team.values()
91 ) / max(len(battle.opponent_team), 1)
92 return (own_hp - opp_hp) * 0.01

95class DoublesGymEnv(Env):
96 """
97 Envoltorio Gymnasium sobre SingleAgentWrapper + MyDoublesEnv.
98
99 Aplana la observación (quita el action_mask) para compatibilidad
100 con SB3 PPO + MlpPolicy.
101
102 El espacio de acciones es MultiDiscrete([107, 107]).
103 SB3 maneja MultiDiscrete nativamente con la política "MlpPolicy".
104
105 Formatos soportados:
106 - "gen9randomdoublesbattle" → equipos aleatorios, sin team_path, SIEMPRE funciona.
107 - "gen9vgc2025regg" (u otro VGC) → requiere team_path con equipo VGC válido.
108
109 team_path puede ser:
110 - None → formatos aleatorios (sin equipo)
111 - un .txt → siempre el mismo equipo (mirror match fijo)
112 - una carpeta → equipo ALEATORIO por combate (cada batalla usa equipos
113 distintos de tu lista; normalmente no-mirror)
114
115 Uso rápido (sin equipo):
116 env = DoublesGymEnv(port=8000)
117
118 Uso VGC con un equipo fijo:
119 env = DoublesGymEnv(port=8000, battle_format="gen9vgc2025regg",
120 team_path="teams/vgc/equipo1.txt")
121
122 Uso VGC con equipos variados (carpeta):
123 env = DoublesGymEnv(port=8000, battle_format="gen9vgc2025regg",
124 team_path="teams/vgc")
125
126 NOTA: gen9doublesou y similares no funcionan (poke-env solo soporta
127 teampreview para formatos VGC en SingleAgentWrapper).
128 """
