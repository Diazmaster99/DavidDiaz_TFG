5"""
6Parche en memoria para un bug de poke-env con los movimientos que CAMBIAN DE ID
7al cambiar de forma:
8
9 - Zacian-Coronado: Iron Head -> Behemoth Blade (behemothblade)
10 - Zamazenta-Coronado: Iron Head -> Behemoth Bash (behemothbash)
11
12El servidor de Showdown reporta el movimiento con el id de la forma
13(behemothblade/behemothbash), pero poke-env guarda en el moveset del Pokémon el
14id BASE (ironhead). Las funciones de conversión orden<->acción de poke-env
15construyen la lista de huecos a partir de ese moveset (con ironhead) y luego la
16comparan con lo que envia/espera el servidor (behemothbash), de modo que fallan:
17
18 - order_to_action (oponente): [m.id for m in mvs].index('behemothbash')
19 -> ValueError: 'behemothbash' is not in list
20 - action_to_order (AGENTE): crea '/choose move ironhead', que el servidor
21 rechaza porque la jugada válida es '/choose move behemothbash'.
22
23Es decir, el bug afecta a AMBOS sentidos: ni el oponente ni el propio agente
24pueden usar Behemoth Blade/Bash (les sale jugada aleatoria ese turno).
25
26Arreglo: 'reconciliar' el moveset conocido -> cuando la variante de forma
27(behemothbash) aparece en available_moves, se usa esa (mismo hueco) en lugar del
28id base (ironhead). Así los ids coinciden con los del servidor y la conversión
29funciona en los dos sentidos. Se aplica monkeypatching a las funciones de
30poke-env al llamar apply(); es idempotente y no altera el comportamiento de
31ningún Pokémon que no tenga uno de estos movimientos de cambio de forma.
32"""

50_FORM_MOVE_FALLBACK = {
51 "behemothblade": "ironhead", # Zacian-Coronado

56def _reconcile(active_mon, available_moves):
57 """Lista de (hasta 4) movimientos conocidos del Pokémon, sustituyendo el
58 movimiento BASE (p.ej. ironhead) por su VARIANTE DE FORMA (behemothbash)
59 cuando está aparece en available_moves. Conserva el orden (= huecos). Si no
60 hay ninguna variante de forma activa, devuelve el moveset tal cual."""
61 known = list(active_mon.moves.values())[:4]
62 by_base = {}
63 for am in available_moves:
64 base = _FORM_MOVE_FALLBACK.get(am.id)
65 if base is not None:
66 by_base[base] = am
67 if not by_base:
68 return known
69 return [by_base.get(m.id, m) for m in known]

121 if action == -2:
122 return DefaultBattleOrder()

134 )

136 avail_ids = [m.id for m in available]

138 known_ids = [m.id for m in known_moves]

140 available

143 )
144 if (action - 7) % 20 // 5 not in range(len(mvs)):
145 raise ValueError(

147 f"in battle {battle.battle_tag} at position {pos} - action "
148 f"specifies a move but the move index {(action - 7) % 20 // 5} "
149 f"is out of bounds for available moves {mvs}!"
150 )
151 order = Player.create_order(
152 mvs[(action - 7) % 20 // 5],
153 move_target=(action.item() - 7) % 5 - 2,
154 mega=(action - 7) // 20 == 1,
155 z_move=(action - 7) // 20 == 2,
156 dynamax=(action - 7) // 20 == 3,
157 terastallize=(action - 7) // 20 == 4,
158 )
159 if not fake and str(order) not in [str(o) for o in battle.valid_orders[pos]]:
160 raise ValueError(
161 f"Invalid action {action} from player {battle.player_username} "
162 f"in battle {battle.battle_tag} at position {pos} - order {order} "
163 f"not in action space {[str(o) for o in battle.valid_orders[pos]]}!"
