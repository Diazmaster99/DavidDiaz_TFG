5"""
6Añade al servidor FORKEADO (pokemon-showdown/) todos los formatos VGC actuales
7del repositorio OFICIAL de Pokémon Showdown que aún no estén en el fork.
8
9Así el fork mantiene sus arreglos de estabilidad (gestión de salas/timeouts)
10y además tiene los formatos VGC al día (por ejemplo gen9vgc2026regi, regf, Bo3...).
11
12Qué hace:
13 1. Descarga config/formats.ts del repo oficial (smogon/pokemon-showdown).
14 2. Extrae las entradas de formato cuyo nombre contiene "VGC".
15 3. Compara por ID con las que ya tiene el fork y añade SOLO las que faltan.
16 4. Reescribe el config/formats.ts del fork.
17
18Después hay que recompilar y reiniciar el servidor:
19 cd pokemon-showdown
20 node build
21 node pokemon-showdown start --no-security
22
23Uso:
24 python patch_server_formats.py
25 python patch_server_formats.py --dry-run (solo muestra qué añadiría)
26"""

42def to_id(name: str) -> str:
43 """Replica toID() de Showdown: minúsculas y solo alfanumérico."""
44 return re.sub(r"[^a-z0-9]", "", name.lower())

47def split_top_level_blocks(text: str) -> list[str]:
48 """
49 Trocea el array de formatos en bloques de nivel superior: cada entrada
50 '{ ... }' indentada con un tab. Devuelve la lista de bloques (texto).
51 """
52 lines = text.split("\n")
53 blocks = []
54 current = None
55 for line in lines:
56 if line == "\t{":
57 current = [line]
58 elif current is not None:
59 current.append(line)
60 if line == "\t}," or line == "\t}":
61 blocks.append("\n".join(current))
62 current = None
63 return blocks

66def block_name(block: str) -> str | None:
67 m = re.search(r'name:\s*"([^"]+)"', block)
68 return m.group(1) if m else None

71def main():
72 parser = argparse.ArgumentParser(description=__doc__)
73 parser.add_argument("--dry-run", action="store_true",
74 help="Solo muestra qué formatos añadiría, sin escribir.")
75 parser.add_argument("--url", default=OFFICIAL_URL,
76 help="URL del formats.ts oficial.")
77 parser.add_argument("--fork", default=str(FORK_FORMATS),
78 help="Ruta al formats.ts del fork.")
79 args = parser.parse_args()
80
81 fork_path = Path(args.fork)
82 if not fork_path.exists():
83 raise FileNotFoundError(
84 f"No existe {fork_path}. Ejecuta desde la raíz del proyecto "
85 "(donde está la carpeta pokemon-showdown/)."
86 )
87
88 print(f"Descargando formatos oficiales de:\n {args.url}")
89 with urllib.request.urlopen(args.url, timeout=30) as resp:
90 official = resp.read().decode("utf-8")
91
92 fork_text = fork_path.read_text(encoding="utf-8")
93 nl = "\r\n" if "\r\n" in fork_text else "\n"
94 # Normalizamos a \n para procesar; restauramos al final.
95 fork_work = fork_text.replace("\r\n", "\n")
96
97 # IDs de formato que ya tiene el fork
98 fork_ids = set()
99 for blk in split_top_level_blocks(fork_work):
100 n = block_name(blk)
101 if n:
102 fork_ids.add(to_id(n))
103
104 # Formatos VGC del oficial que faltan en el fork
105 official_blocks = split_top_level_blocks(official.replace("\r\n", "\n"))
106 to_add = []
107 seen = set()
108 for blk in official_blocks:
109 n = block_name(blk)
110 if not n or "vgc" not in n.lower():
111 continue
112 fid = to_id(n)
113 if fid in fork_ids or fid in seen:
114 continue
115 seen.add(fid)
116 to_add.append((n, fid, blk))
117
118 if not to_add:
119 print("\nEl fork ya tiene todos los formatos VGC del oficial. Nada que hacer.")
120 return
121
122 print(f"\nFormatos VGC a añadir ({len(to_add)}):")
123 for n, fid, _ in to_add:
124 print(f" + {n} (id: {fid})")
125
126 if args.dry_run:
127 print("\n[dry-run] No se ha escrito nada.")
128 return
129
130 # Insertar los bloques nuevos tras la entrada ancla del fork
131 anchor_idx = fork_work.find(f'name: "{ANCHOR_NAME}"')
132 if anchor_idx == -1:
133 # Fallback: insertar antes del cierre del array "\n];"
134 insert_at = fork_work.rfind("\n];")
135 if insert_at == -1:
136 raise RuntimeError("No se encontró punto de inserción en el formats.ts del fork.")
137 block_end = insert_at
138 else:
139 # Cierre del bloque ancla: primer "\n\t}," tras el ancla
140 close = fork_work.find("\n\t},", anchor_idx)
141 block_end = close + len("\n\t},") if close != -1 else fork_work.rfind("\n];")
142
143 insertion = "\n" + "\n".join(blk for _, _, blk in to_add)
144 new_text = fork_work[:block_end] + insertion + fork_work[block_end:]
145
146 # Restaurar fin de línea original
147 if nl == "\r\n":
148 new_text = new_text.replace("\n", "\r\n")
149 fork_path.write_text(new_text, encoding="utf-8", newline="")
150
151 print(f"\nAñadidos {len(to_add)} formatos a {fork_path}")
152 print("\nAhora recompila y reinicia el servidor:")
153 print(" cd pokemon-showdown")
154 print(" node build")
155 print(" node pokemon-showdown start --no-security")
