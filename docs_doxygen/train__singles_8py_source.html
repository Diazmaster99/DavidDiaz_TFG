44def checkpoint_steps(path) -> int | None:
45 """
46 Extrae el número de pasos del nombre de un checkpoint.
47 Devuelve None para ficheros sin número (por ejemplo 'final.zip').
48 """
49 cleaned = path.stem.replace("model_", "").replace("_steps", "")
50 try:
51 return int(cleaned)
52 except ValueError:
53 return None

73def train(
74 run_id: int,
75 port: int,
76 device: str,
77 battle_format: str,
78 total_steps: int,
79 save_interval: int,
80 opponent_type: str,
81 policy: str = "mlp",
82 tag: str = "",
83):
84 set_seed(run_id)
85 check_server(port)
86
87 # Separar modelos por formato Y por política (para comparar mlp vs transformer)
88 # 'tag' permite diferenciar modelos/runs (p.ej. tag="selfplay") sin mezclarlos
89 # con el mismo seed entrenado contra otro oponente.
90 seed_dir = f"seed{run_id}" + (f"_{tag}" if tag else "")
91 save_dir = Path("models/singles") / battle_format / policy / seed_dir
92 log_dir = Path("logs/singles") / battle_format / policy
93 save_dir.mkdir(parents=True, exist_ok=True)
94 log_dir.mkdir(parents=True, exist_ok=True)
95
96 print(f"Creando entorno {battle_format} contra {opponent_type} (policy={policy})...")
97 # SinglesGymEnv ya crea su propio oponente (sin conexión al servidor)
98 env = SinglesGymEnv(
99 port=port,
100 battle_format=battle_format,
101 log_level=40,
102 opponent_type=opponent_type,
103 )
104
105 # Reanudar desde checkpoint si existe. Probamos de MAYOR a menor número de
106 # pasos y cargamos el primero compatible (ignora modelos PPO antiguos sin
107 # action masking, que MaskablePPO no puede cargar).
108 num_saved = 0
109 model = None
110 numbered = sorted(
111 [
112 (p, checkpoint_steps(p))
113 for p in save_dir.glob("*.zip")
114 if checkpoint_steps(p) is not None
115 ],
116 key=lambda t: t[1],
117 reverse=True,
118 )
119 for cand, steps in numbered:
120 try:
121 model = MaskablePPO.load(str(cand), env=env, device=device)
122 model.num_timesteps = steps
123 num_saved = steps
124 print(f"Reanudando desde {cand} ({steps} pasos)")
125 break
126 except Exception as e:
127 print(f"AVISO: {cand.name} no es compatible ({type(e).__name__}); "
128 "probando un checkpoint anterior...")
129 model = None
130
131 if model is None:
132 from src.policy import make_policy_kwargs
133 print(f"Creando nuevo modelo MaskablePPO (policy={policy}, action masking)...")
134 model = MaskablePPO(
135 "MlpPolicy",
136 env,
137 learning_rate=lambda p: 3e-4 * p, # decae a cero al final
138 n_steps=2048,
139 batch_size=256,
140 n_epochs=10,
141 gamma=0.99,
142 gae_lambda=0.95,
143 clip_range=0.2,
144 ent_coef=0.01,
145 policy_kwargs=make_policy_kwargs(policy),
146 verbose=1,
147 tensorboard_log=str(log_dir),
148 device=device,
149 )
150
151 # SELF-PLAY: el rival juega con la propia política que se está entrenando.
152 if opponent_type == "self":
153 env.set_opponent_model(model)
154 print("SELF-PLAY activado: el rival usa la política actual del agente.")
155
156 checkpoint_cb = CheckpointCallback(
157 save_freq=save_interval,
158 save_path=str(save_dir),
159 name_prefix="model",
160 verbose=1,
161 )
162 # Registra eval/win_rate en TensorBoard (winrate móvil de las últimas batallas)
163 winrate_cb = WinRateCallback(log_freq=5000, window=200)
164
165 remaining = total_steps - num_saved
166 if remaining <= 0:
167 print(f"Ya se han completado {num_saved} >= {total_steps} pasos.")
168 else:
169 print(f"Entrenando {remaining} pasos (total objetivo: {total_steps})...")
170 model.learn(
171 total_timesteps=remaining,
172 callback=[checkpoint_cb, winrate_cb],
173 reset_num_timesteps=False,
174 tb_log_name=f"singles_seed{run_id}" + (f"_{tag}" if tag else ""),
175 )
176 # Guardar modelo final
177 final_path = save_dir / "final"
178 model.save(str(final_path))
179 print(f"Modelo final guardado en {final_path}.zip")
180
181 env.close()
182 print("Entrenamiento completado.")
