5"""
6Enfrenta dos agentes entre sí en el servidor local y reporta el resultado.
7
8Cada lado (--a y --b) puede ser:
9 - Un modelo entrenado: mlp | transformer
10 - Un baseline de poke-env: random | maxpower | heuristic
11
12Se puede medir, por ejemplo:
13 - MLP vs Transformer
14 - tu modelo vs RandomPlayer / MaxBasePowerPlayer / SimpleHeuristicsPlayer
15 - dos seeds del mismo tipo
16
17Los agentes RL usan la misma observación (src/features.py), por lo que
18arquitecturas distintas pueden enfrentarse sin problema.
19
20Uso (dobles VGC, tu MLP vs el heurístico de poke-env):
21 python battle_models.py --mode doubles --format gen9vgc2026regi \\
22 --team teams/vgc/I1.txt --a mlp --b heuristic -n 50
23
24Uso (singles, transformer vs MaxDamage):
25 python battle_models.py --mode singles --a transformer --b maxpower -n 50
26
27Requiere el servidor local corriendo:
28 node pokemon-showdown start --no-security
29"""

72def load_model(mode: str, battle_format: str, policy: str, run_id: int, tag: str = ""):
73 seed_dir = f"seed{run_id}" + (f"_{tag}" if tag else "")
74 model_dir = Path(f"models/{mode}") / battle_format / policy / seed_dir
75 path = find_latest_model(model_dir)
76 if path is None:
77 raise FileNotFoundError(
78 f"No se encontró modelo en {model_dir}\n"
79 f"Entrena con: python train_{mode}.py --format {battle_format} "
80 f"--policy {policy}" + (f" --tag {tag}" if tag else "")
81 )
82 return MaskablePPO.load(str(path)), path

85def make_agent(spec, mode, battle_format, team_builder, server, run_id, name, tag=""):
86 """Crea un jugador: modelo RL (mlp/transformer) o baseline de poke-env."""
87 common = dict(
88 battle_format=battle_format,
89 server_configuration=server,
90 log_level=40,
91 max_concurrent_battles=1,
92 account_configuration=AccountConfiguration(name, None),
93 )
94 if mode == "doubles":
95 common["team"] = team_builder
96
97 if spec in RL_SPECS:
98 model, path = load_model(mode, battle_format, spec, run_id, tag)
99 cls = RLDoublesPlayer if mode == "doubles" else RLSinglesPlayer
100 suffix = f"_{tag}" if tag else ""
101 return cls(model=model, **common), f"{spec} (seed{run_id}{suffix})"
102
103 base_cls = BASELINE_CLS[spec]
104 return base_cls(**common), f"baseline:{spec}"
