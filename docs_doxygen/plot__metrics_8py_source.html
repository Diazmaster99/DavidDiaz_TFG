63def _fmt_steps(x, _pos):
64 """Formatea el eje X de pasos: 0, 200k, 400k, ..., 1M, 1.5M (sin '1e6')."""
65 if x >= 1_000_000:
66 v = x / 1_000_000
67 return f"{v:.0f}M" if abs(v - round(v)) < 1e-9 else f"{v:.1f}M"
68 if x >= 1_000:
69 v = x / 1_000
70 return f"{v:.0f}k" if abs(v - round(v)) < 1e-9 else f"{v:.1f}k"
71 return f"{int(round(x))}"

74def find_runs(logdir: Path) -> dict:
75 """Devuelve {etiqueta: carpeta} por cada directorio que contenga event files."""
76 runs = {}
77 for ev in logdir.rglob("events.out.tfevents.*"):
78 d = ev.parent
79 if d in runs.values():
80 continue
81 try:
82 rel = d.relative_to(logdir)
83 except ValueError:
84 rel = Path(d.name)
85 # ignora carpetas archivadas/ocultas (componentes que empiezan por _ o .)
86 if any(part.startswith(("_", ".")) for part in rel.parts):
87 continue
88 label = str(rel).replace("\\", "/")
89 if label in (".", ""):
90 label = logdir.name
91 runs[label] = d
92 return runs

95def load_scalar(run_dir: Path, tag: str):
96 """Carga (steps, values) de un escalar, fusionando todos los event files
97 de la carpeta, ordenando por step y quedándose con el último valor de cada
98 step (las reanudaciones generan varios ficheros con steps solapados)."""
99 acc = EventAccumulator(str(run_dir), size_guidance={"scalars": 0})
100 acc.Reload()
101 if tag not in acc.Tags().get("scalars", []):
102 return None, None
103 by_step = {}
104 for e in acc.Scalars(tag):
105 by_step[e.step] = e.value # el último valor escrito para ese step gana
106 steps = sorted(by_step)
107 values = [by_step[s] for s in steps]
108 return steps, values

111def ema(values, alpha):
112 """Suavizado exponencial estilo TensorBoard. alpha en [0, 1)."""
113 if not values or alpha <= 0:
114 return list(values)
115 out = []
116 acc = values[0]
117 for v in values:
118 acc = alpha * acc + (1 - alpha) * v
119 out.append(acc)
120 return out
