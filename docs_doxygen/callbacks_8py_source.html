14class WinRateCallback(BaseCallback):
15 """
16 Registra en TensorBoard la TASA DE VICTORIAS móvil (eval/win_rate) sobre las
17 últimas 'window' batallas terminadas.
18
19 La victoria/derrota se deduce de la RECOMPENSA terminal:
20 +1 -> victoria, -1 -> derrota, ~0 -> combate abandonado (no cuenta).
21 Así es robusto incluso si el entorno se reconstruye (no depende de los
22 contadores internos del agente, que se reinician al reconstruir).
23
24 :param log_freq: cada cuántos steps se registra el winrate (0 = nunca).
25 :param window: número de batallas recientes sobre las que se promedia.
26 """
27
28 def __init__(self, log_freq: int = 5000, window: int = 200):
29 super().__init__()
30 self.log_freq = log_freq
31 self._results: deque[float] = deque(maxlen=window)
32
33 def _on_step(self) -> bool:
34 dones = self.locals.get("dones")
35 rewards = self.locals.get("rewards")
36 if dones is not None and rewards is not None:
37 for done, reward in zip(dones, rewards):
38 if done:
39 if reward > 0.5:
40 self._results.append(1.0)
41 elif reward < -0.5:
42 self._results.append(0.0)
43 # reward ~0 -> combate abandonado/empate: no se cuenta
44 if self.log_freq and self.n_calls % self.log_freq == 0 and self._results:
45 self.logger.record(
46 "eval/win_rate", sum(self._results) / len(self._results)
47 )
48 self.logger.record("eval/battles_contadas", len(self._results))
49 return True
